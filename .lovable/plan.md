## Part 12: Conversion Tracking Setup

### Google Analytics 4 Events

```javascript
// Event: View Product
gtag('event', 'view_item', {
  currency: 'USD',
  value: product.price,
  items: [{
    item_id: product.id,
    item_name: product.name,
    item_category: 'eSIM'
  }]
});

// Event: Begin Checkout
gtag('event', 'begin_checkout', {
  currency: 'USD',
  value: cart.total,
  coupon: discount_code
});

// Event: Purchase Complete
gtag('event', 'purchase', {
  transaction_id: order.id,
  value: order.amount,
  currency: 'USD',
  coupon: order.discount_code
});
```

### Meta Pixel Events

```javascript
// ViewContent
fbq('track', 'ViewContent', {
  content_name: product.name,
  content_type: 'product',
  content_ids: [product.id],
  value: product.price,
  currency: 'USD'
});

// InitiateCheckout
fbq('track', 'InitiateCheckout', {
  value: cart.total,
  currency: 'USD',
  num_items: cart.items.length
});

// Purchase
fbq('track', 'Purchase', {
  value: order.amount,
  currency: 'USD',
  content_type: 'product',
  content_ids: order.product_ids
});
```
